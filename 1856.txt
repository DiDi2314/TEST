#include <stdio.h>
 
void main(){
    int n, m, count=0;
    scanf("%d %d", &n, &m); //
    for(int i=0; i<n; i++){
        if(i%2==0){
            for(int j=0; j<m; j++){
                printf("%d ", ++count);
            }
            printf("\n");
            count+=m;       
        }
        else if(i%2==1){
            for(int j=0; j<m; j++)
                printf("%d ", count--);
            count+=m;
            printf("\n");
        }
    }
}
